// Function to determine performance based on fuzzy rules
public static String fuzzyPerformanceEvaluation(int attendance, int marks) {
String attLevel = fuzzyAttendance(attendance);
String marksLevel = fuzzyMarks(marks);
String performance = "Poor"; // Default performance
// Fuzzy rules implementation
if (attLevel.equals("High") && marksLevel.equals("High")) {
performance = "Excellent";
} else if ((attLevel.equals("Medium") || attLevel.equals("High")) &&
(marksLevel.equals("Medium") || marksLevel.equals("High"))) {
performance = "Good";
} else if (attLevel.equals("Low") && marksLevel.equals("Medium")) {
performance = "Average";
} else if (attLevel.equals("Medium") && marksLevel.equals("Low")) {
performance = "Poor";
}
return performance;
}
// Main method for user input and execution
public static void main(String[] args) {
Scanner scanner = new Scanner(System.in);
// User input for attendance and marks
System.out.print("Enter Attendance Percentage: ");
int attendance = scanner.nextInt();
System.out.print("Enter Marks Percentage: ");
int marks = scanner.nextInt();
// Call fuzzy evaluation method
String performance = fuzzyPerformanceEvaluation(attendance, marks);
// Output result
System.out.println("Performance Evaluation Result: " + performance);
scanner.close();
}
}
